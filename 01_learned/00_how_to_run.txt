for backend
  in root of the project

    bundle install
    rails db:setup

for frontend
  in ./client

    create file
      client/.env.development

      with this content for now (is not that sensitive for now)
        VITE_APP_ENV=development
        VITE_BASE_URL=http://localhost:4000
        VITE_API_URL=http://localhost:4000/api/v1

    npm install

    npm run dev
